//Find the srategy to pick chips if you can pick 'moves' no. of chips.// You lose when you cant pick up any more chips// its a 2 player game played alternativly#include<bits/stdc++.h>using namespace std;vector<int> moves = {2,3,5};int dp[100100];int rec(int x){    if(dp[x]!=-1){        return dp[x];    }    int ans = 0;    for(auto v:moves){        if(x>=v && rec(x-v)==0){            ans = 1;            break;        }    }    return dp[x]=ans;}void solve(){    int n;    cin>>n;    memset(dp,-1,sizeof(dp));    cout<<rec(n)<<endl;}    signed main(){    ios_base::sync_with_stdio(0);cin.tie(0);cout.tie(0);    //int _t;cin>>_t;while(_t--)    solve();}
